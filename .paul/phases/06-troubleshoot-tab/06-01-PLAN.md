<objective>
## Goal
Add a "Fix It" fourth tab to the bottom nav with 6 expandable symptom cards, each with 1–3 ordered fixes — a trail-side cheat sheet riders can open when something feels wrong.

## Purpose
This is the key Phase 6 deliverable. A rider finishes a descent that felt wrong, opens the app, taps the symptom, and gets an ordered action list in seconds — without knowing damping theory.

## Output
- 4th nav button ("Fix It", wrench icon) between Saved and Settings
- `#tab-fixit` panel with accordion symptom cards
- `SYMPTOM_FIXES` data array (6 symptoms × 2–3 fixes each)
- `renderFixitTab()` + `toggleSymptom(id)` functions
- CSS for cards, accordion, numbered fix list
</objective>

    **2. Add `renderFixitTab()` function** — after `switchTab()` function (~line 3916):

    ```js
    let fixitRendered = false;
    function renderFixitTab() {
      if (fixitRendered) return;
      fixitRendered = true;
      const inner = document.getElementById('fixit-inner');
      inner.innerHTML = `
        <div class="fixit-header">
          <h2 class="fixit-title">Fix It</h2>
          <p class="fixit-subtitle">Tap a symptom to see what to adjust</p>
        </div>
        <div class="fixit-cards">
          ${SYMPTOM_FIXES.map(s => `
            <div class="fixit-card" id="fixit-card-${s.id}">
              <button class="fixit-card-trigger" onclick="toggleSymptom('${s.id}')">
                <i class="fa-solid ${s.icon} fixit-card-icon"></i>
                <span class="fixit-card-label">${s.label}</span>
                <i class="fa-solid fa-chevron-right fixit-chevron" id="fixit-chev-${s.id}"></i>
              </button>
              <div class="fixit-card-body" id="fixit-body-${s.id}">
                <ol class="fixit-fix-list">
                  ${s.fixes.map(f => `
                    <li class="fixit-fix-item">
                      <div class="fixit-fix-title">${f.title}</div>
                      <div class="fixit-fix-detail">${f.detail}</div>
                    </li>
                  `).join('')}
                </ol>
              </div>
            </div>
          `).join('')}
        </div>
      `;
    }

    let openSymptomId = null;
    function toggleSymptom(id) {
      const body = document.getElementById('fixit-body-' + id);
      const chev = document.getElementById('fixit-chev-' + id);
      const isOpen = body.classList.contains('open');

      // Close currently open card
      if (openSymptomId && openSymptomId !== id) {
        document.getElementById('fixit-body-' + openSymptomId).classList.remove('open');
        document.getElementById('fixit-chev-' + openSymptomId).classList.remove('open');
      }

      body.classList.toggle('open', !isOpen);
      chev.classList.toggle('open', !isOpen);
      openSymptomId = isOpen ? null : id;
    }
    ```
