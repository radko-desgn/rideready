<objective>
## Goal
Add a bracketing history log to each saved suspension setup — riders can record the PSI they tried after a ride and how it felt, building a personal tuning history over time.

## Purpose
Riders currently get a calculated starting PSI and save it — but they have no way to track what they actually ran on the trail and whether it worked. A bracketing log closes this loop: try the recommended 85 PSI, come back and log "too stiff," drop to 80, log "perfect." Over a few rides they dial in their setup and can see the progression.

## Output
- Supabase `saves` table updated with a `history` JSONB column
- History modal (new `#history-modal-backdrop`) on each suspension save card
- Each entry stores: date, PSI tried, component (fork/shock/both), feel (Perfect / Too stiff / Too soft)
- `addHistoryEntry(saveId, entry)` function updates Supabase and refreshes the modal list
</objective>

    ```js
    function openHistoryModal(saveId) {
      const save = cachedSaves.find(s => s.id === saveId);
      if (!save) return;
      document.getElementById('history-modal-title').textContent = save.name || 'Ride History';
      document.getElementById('history-modal-backdrop').dataset.saveId = saveId;
      renderHistoryModalBody(save);
      document.getElementById('history-modal-backdrop').classList.add('open');
    }

    function closeHistoryModal() {
      document.getElementById('history-modal-backdrop').classList.remove('open');
    }

    function renderHistoryModalBody(save) {
      const entries = Array.isArray(save.history) ? save.history : [];
      const hasFork = !!save.fork_model;
      const hasShock = !!save.shock_model;
      const defaultComponent = hasFork && hasShock ? 'both' : hasFork ? 'fork' : 'shock';

      const entriesHTML = entries.length === 0
        ? `<p class="history-empty">No rides logged yet.<br>Log your first ride below.</p>`
        : `<div class="history-log">${entries.slice().reverse().map(e => `
            <div class="history-entry">
              <span class="history-entry-date">${relativeDate(e.date)}</span>
              <span class="history-entry-psi">${e.psi} PSI</span>
              <span class="history-entry-component">${{ fork: 'Fork', shock: 'Shock', both: 'Fork & Shock' }[e.component] || ''}</span>
              <span class="history-feel ${e.feel}">${{ perfect: 'Perfect', 'too-stiff': 'Too stiff', 'too-soft': 'Too soft' }[e.feel] || e.feel}</span>
            </div>`).join('')}
          </div>`;

      document.getElementById('history-modal-body').innerHTML = `
        ${entriesHTML}
        <div class="history-add-form">
          <p class="history-add-title">Log this ride</p>
          <div class="history-component-row">
            ${hasFork ? `<button class="history-component-btn${defaultComponent==='fork'||defaultComponent==='both'?' selected':''}" data-component="fork" onclick="selectHistoryComponent(this)">Fork</button>` : ''}
            ${hasShock ? `<button class="history-component-btn${defaultComponent==='shock'?' selected':''}" data-component="shock" onclick="selectHistoryComponent(this)">Shock</button>` : ''}
            ${hasFork && hasShock ? `<button class="history-component-btn${defaultComponent==='both'?' selected':''}" data-component="both" onclick="selectHistoryComponent(this)">Both</button>` : ''}
          </div>
          <div class="history-psi-row">
            <label class="history-psi-label">PSI tried</label>
            <input id="history-psi-input" class="history-psi-input" type="number" min="20" max="300" placeholder="e.g. 85">
          </div>
          <div class="history-feel-row">
            <button class="history-feel-btn" data-feel="perfect" onclick="selectHistoryFeel(this)">Perfect</button>
            <button class="history-feel-btn" data-feel="too-stiff" onclick="selectHistoryFeel(this)">Too stiff</button>
            <button class="history-feel-btn" data-feel="too-soft" onclick="selectHistoryFeel(this)">Too soft</button>
          </div>
          <button class="history-save-btn" onclick="submitHistoryEntry()" id="history-save-btn">Save ride</button>
        </div>`;
    }

    function selectHistoryComponent(btn) {
      btn.closest('.history-component-row').querySelectorAll('.history-component-btn').forEach(b => b.classList.remove('selected'));
      btn.classList.add('selected');
    }

    function selectHistoryFeel(btn) {
      btn.closest('.history-feel-row').querySelectorAll('.history-feel-btn').forEach(b => b.classList.remove('selected'));
      btn.classList.add('selected');
    }

    async function submitHistoryEntry() {
      const backdrop = document.getElementById('history-modal-backdrop');
      const saveId = backdrop.dataset.saveId;
      const psi = parseInt(document.getElementById('history-psi-input').value);
      const feelBtn = document.querySelector('.history-feel-btn.selected');
      const componentBtn = document.querySelector('.history-component-btn.selected');

      if (!psi || psi < 20 || !feelBtn) return;

      const entry = {
        id: crypto.randomUUID(),
        date: new Date().toISOString(),
        psi,
        feel: feelBtn.dataset.feel,
        component: componentBtn ? componentBtn.dataset.component : 'fork'
      };

      const save = cachedSaves.find(s => s.id === saveId);
      if (!save) return;

      const updatedHistory = [...(Array.isArray(save.history) ? save.history : []), entry];

      const { error } = await supabaseClient
        .from('saves')
        .update({ history: updatedHistory })
        .eq('id', saveId);

      if (error) { console.error('History save error:', error); return; }

      // Update cachedSaves
      save.history = updatedHistory;

      // Re-render modal body
      renderHistoryModalBody(save);
    }
    ```

    **Step 3 — Add `relativeDate(isoString)` helper** near other utility functions:
    ```js
    function relativeDate(iso) {
      const diff = Date.now() - new Date(iso).getTime();
      const days = Math.floor(diff / 86400000);
      if (days === 0) return 'Today';
      if (days === 1) return 'Yesterday';
      if (days < 7) return `${days} days ago`;
      if (days < 30) return `${Math.floor(days/7)}w ago`;
      return new Date(iso).toLocaleDateString('en-GB', { day: 'numeric', month: 'short' });
    }
    ```
